TabelaSidra1 <- melt(TabelaSidra, id.vars = "Ano")
names(TabelaSidra1)[2:3] <- c("Mês", "Nascimentos")

graficosidra1<-ggplot(TabelaSidra1, aes(Mês, Ano)) +
  geom_tile(aes(fill = Nascimentos), color = "white") +
  scale_fill_gradient(low = "white", high = "#000066") +
  ylab("Ano") +
  xlab("Mês") +
  theme(legend.title = element_text(size = 10),
        legend.text = element_text(size = 12),
        plot.title = element_text(size=16),
        axis.title=element_text(size=14,face="bold"),
        axis.text.x = element_text(angle = 90, hjust = 1)) +
  labs(fill = "Nascimentos") + ggtitle("Mapa de calor - Nascidos Vivos")

graficosidra2<-ggplot(TabelaSidra2,aes(x = Anos_Meses, y = Nascimentos)) +
  geom_line(col="green") + theme(
    panel.grid.major = element_line(size = 0.25, linetype = 'solid',
                                    colour = "#3F7B38"), 
    panel.grid.minor = element_blank(),
    plot.background=element_rect(fill = "#E2E2E2"),
    panel.background = element_rect(fill = '#181818'),
    legend.key = element_rect(color = "gray", fill = "black"),
    legend.title = element_text(color = "white"),
    legend.text = element_text(color = "white")) +
  xlab("Meses e Anos") + geom_vline(xintercept=c(2003.12,2004.12,2005.12,2006.12,
                                                 2007.12,2008.12,2009.12,2010.12,
                                                 2011.12,2012.12,2013.12,2014.12,
                                                 2015.12,2016.12,2017.12),
                                    linetype = "dashed", color = "#3F7B38" ) +
  labs(title = "Nascidos vivos por Mês de cada Ano (2003 - 2017)",
         subtitle = "Linhas pontilhadas representam o mês de dezembro de cada ano,
       onde ocorre o salto para o mês de janeiro do ano seguinte",
         caption = "Fonte: SIDRA (IBGE)") +
  theme(plot.title = element_text(hjust = 0.5), plot.subtitle = element_text(hjust = 0.5))
